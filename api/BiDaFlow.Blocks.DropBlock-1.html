Examples
var sourceBlock = new BufferBlock<int>();
for (int i = 1; i <= 5; i++)
    sourceBlock.Post(i);
sourceBlock.Complete();

// Drops even values
sourceBlock.LinkTo(new DropBlock<int>(), x => x % 2 == 0);

await sourceBlock
    .RunWith(new ActionBlock<int>(x => Console.WriteLine(x)))
    .Completion;
